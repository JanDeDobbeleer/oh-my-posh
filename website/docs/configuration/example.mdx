blocks:
- type: rprompt

  - foreground: "#007ACC"

    properties:
      time_format: '15:04:05'
    style: plain
    type: time

  newline: true
  segments:
  - background: "#ffb300"
    foreground: "#ffffff"
    leading_diamond: "\ue0b6"
    template: " {{ .UserName }} "
    style: diamond
    trailing_diamond: "\ue0b0"
    type: session
  - background: "#61AFEF"
    foreground: "#ffffff"
    powerline_symbol: "\ue0b0"
    template: " {{ .Path }} "
    properties:
      exclude_folders:
      - "/super/secret/project"
      style: folder
    style: powerline
    type: path
  - background: "#2e9599"
    background_templates:
    - "{{ if or (.Working.Changed) (.Staging.Changed) }}#f36943{{ end }}"
    - "{{ if and (gt .Ahead 0) (gt .Behind 0) }}#a8216b{{ end }}"
    - "{{ if gt .Ahead 0 }}#35b5ff{{ end }}"
    - "{{ if gt .Behind 0 }}#f89cfa{{ end }}"
